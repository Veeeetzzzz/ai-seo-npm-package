export class SchemaDetector {
  static detect(parsedContent) {
    const types = [];
    
    // Check for Product indicators
    if (this._hasProductIndicators(parsedContent)) {
      types.push({ 
        type: 'Product', 
        confidence: this._calculateConfidence(parsedContent, 'Product')
      });
    }
    
    // Check for Article indicators
    if (this._hasArticleIndicators(parsedContent)) {
      types.push({ 
        type: 'Article', 
        confidence: this._calculateConfidence(parsedContent, 'Article')
      });
    }
    
    // Sort by confidence
    return types.sort((a, b) => b.confidence - a.confidence);
  }
